#### 网站登陆和打码平台使用

```python
# encoding = utf-8
from selenium import webdriver
import time
import requests

driver = webdriver.Chrome()

driver.get("https://www.tanwan.com/index.php?act=login")

# 下载验证码
response = requests.get("https://hd.tanwan.com/include/chkcode.inc.php?w=80&h=30")
with open("./captcha.png", "wb") as f:
    f.write(response.content)

driver.find_element_by_id("login_user").send_keys("18819268053")
driver.find_element_by_id("login_pass").send_keys("123456")

# 验证码识别
url = '打码平台api'
heads={'Content-type':'image/jpeg'}
t = open('./captcha.png','rb')
res = requests.post(url=url,data=t,headers=heads)
print(res.text)

driver.find_element_by_id("login_yzm").send_keys(res.text)
driver.find_element_by_class_name("log-ttn-dl").click()

cookies = driver.get_cookies()
print(cookies)

driver.quit()
```
